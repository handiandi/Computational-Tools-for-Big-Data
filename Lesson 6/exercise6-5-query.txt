MATCH (customer)--(orders)--(products : Product) 
WITH collect(DISTINCT customer.customerID) as Cus, products
where products.productID = '7' 
MATCH (Product)--(orders)
WITH DISTINCT Product.productName as pros, orders, Cus
WHERE orders.customerID IN Cus 
RETURN collect(DISTINCT pros) as Products, length(collect(distinct pros)) as NumberOfProducts
